\newcommand\dangersign[1][3ex]{
  \renewcommand\stacktype{L}
  \scaleto{\stackon[.8pt]{\color{red}$\triangle$}{\tiny\bfseries !}}{#1}
}

\newcommand\infosign[1][3ex]{
  \renewcommand\stacktype{L}
  \scaleto{\stackon[-1.5pt]{\scalebox{.3}{\bfseries i}}{\color{blue}$\circ$}}{#1}
}

    \begin{list}{}{

        \leftmargin
    }
    \item[
        \dangersign

}

    \end{list}
    \end{mdframed}
    \par
}
